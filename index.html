/** * BOLT Dashboard — Private Data Proxy (Cloudflare Worker) * --------------------------------------------------------------- * Replaces public CORS proxies (corsproxy.io / allorigins.win) with a * proxy you control: faster, not shared/rate-limited by strangers, and * cached briefly so multiple tabs/refreshes don't hammer Yahoo/Stooq/RSS. * * DEPLOY (about 5 minutes, free tier is far more than enough): * 1. Go to https://dash.cloudflare.com -> Workers & Pages -> Create -> Worker * 2. Delete the placeholder code and paste this entire file in. * 3. (Optional but recommended) Click "Settings" and set an environment * variable ALLOWED_ORIGIN to the origin you'll load the dashboard from * (e.g. "https://yourname.github.io" or "null" if opening the HTML * file directly from disk). Leave unset to allow any origin. * 4. Click "Deploy". You'll get a URL like: * https://bolt-proxy..workers.dev * 5. In the BOLT dashboard's "Data & Integrations" panel, set the * "Private Data Proxy URL" field to: * https://bolt-proxy..workers.dev/?url= * (keep the trailing "?url=" — the dashboard appends the encoded * target URL after it automatically) * * This Worker only fetches whatever URL the dashboard asks for (Yahoo * Finance, Stooq, and the RSS feeds already hardcoded in the dashboard). * It does not accept arbitrary instructions from page content — it is a * plain fetch-and-relay proxy with a short cache. */ const CACHE_TTL_SECONDS = 20; // short cache so quotes/news stay close to real-time but repeat calls are cheap export default { async fetch(request, env, ctx) { const requestOrigin = request.headers.get('Origin') || ''; const allowedOrigin = env.ALLOWED_ORIGIN || '*'; if (request.method === 'OPTIONS') { return new Response(null, { headers: corsHeaders(allowedOrigin, requestOrigin) }); } const incoming = new URL(request.url); const target = incoming.searchParams.get('url'); if (!target) { return new Response('Missing "url" query parameter.', { status: 400, headers: corsHeaders(allowedOrigin, requestOrigin) }); } let targetUrl; try { targetUrl = new URL(target); } catch { return new Response('Invalid target URL.', { status: 400, headers: corsHeaders(allowedOrigin, requestOrigin) }); } // Only allow http(s) targets — blocks anything trying to hit internal/other schemes. if (targetUrl.protocol !== 'http:' && targetUrl.protocol !== 'https:') { return new Response('Only http/https targets are allowed.', { status: 400, headers: corsHeaders(allowedOrigin, requestOrigin) }); } const cache = caches.default; const cacheKey = new Request(targetUrl.toString(), request); let response = await cache.match(cacheKey); if (!response) { try { const upstream = await fetch(targetUrl.toString(), { headers: { 'User-Agent': 'Mozilla/5.0 (compatible; BOLT-Dashboard-Proxy/1.0)' }, cf: { cacheTtl: CACHE_TTL_SECONDS, cacheEverything: true } }); const body = await upstream.text(); response = new Response(body, { status: upstream.status, headers: { 'Content-Type': upstream.headers.get('Content-Type') || 'text/plain; charset=utf-8' } }); response.headers.set('Cache-Control', `public, max-age=${CACHE_TTL_SECONDS}`); ctx.waitUntil(cache.put(cacheKey, response.clone())); } catch (err) { return new Response('Upstream fetch failed: ' + err.message, { status: 502, headers: corsHeaders(allowedOrigin, requestOrigin) }); } } const finalHeaders = new Headers(response.headers); const cors = corsHeaders(allowedOrigin, requestOrigin); for (const [k, v] of Object.entries(cors)) finalHeaders.set(k, v); return new Response(response.body, { status: response.status, headers: finalHeaders }); } }; function corsHeaders(allowedOrigin, requestOrigin) { const origin = allowedOrigin === '*' ? '*' : (requestOrigin === allowedOrigin ? requestOrigin : allowedOrigin); return { 'Access-Control-Allow-Origin': origin, 'Access-Control-Allow-Methods': 'GET, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', 'Vary': 'Origin' }; }
